//Пример структуры
src/
├── assets/
├── components/           # Глобальные, общедоступные компоненты (ui-kit)
│   └── ui/
├── composables/          # Глобальные композаблы (например, useApi)
├── features/             # Папка для каждой крупной функциональности приложения
│   ├── auth/             # Вся логика, связанная с аутентификацией
│   │   ├── components/   # Компоненты, специфичные для auth
│   │   ├── composables/  # Композаблы, специфичные для auth (useAuth)
│   │   ├── stores/       # Стор (Pinia) для управления состоянием auth
│   │   └── utils/        # Вспомогательные функции для auth
│   └── dashboard/        # Другая фича - dashboard
│       ├── components/
│       ├── composables/
│       └── stores/
├── layouts/              # Компоненты-макеты (шапка, боковое меню, подвал)
├── pages/                # Страницы (аналогично views)
├── router/
├── stores/               # Глобальные хранилища (если не используются в features/)
├── utils/
├── App.vue
└── main.js
